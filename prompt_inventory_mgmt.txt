// ────────────────────────────────
// Box: 'products' (TypeId: 0)
// ────────────────────────────────
@HiveType(typeId: 0)
class ProductModel extends HiveObject {
  @HiveField(0)  late String id;
  @HiveField(1)  late String name;
  @HiveField(2)  late String description;
  @HiveField(3)  late String sku;
  @HiveField(4)  String? barcode;
  @HiveField(5)  late String categoryId;
  @HiveField(6)  late double costPrice;
  @HiveField(7)  late double sellingPrice;
  @HiveField(8)  late int currentStock;
  @HiveField(9)  late int minimumStock;
  @HiveField(10) late int maximumStock;
  @HiveField(11) late String unit;         // 'pcs','kg','liters','boxes'
  @HiveField(12) String? imagePath;
  @HiveField(13) String? supplierId;
  @HiveField(14) DateTime? expiryDate;
  @HiveField(15) late DateTime createdAt;
  @HiveField(16) late DateTime updatedAt;
  @HiveField(17) String? batchNumber;
  @HiveField(18) String? warehouseLocation;
  @HiveField(19) late List<String> tags;
  @HiveField(20) late bool isActive;
  @HiveField(21) double? weight;
  @HiveField(22) Map<String, dynamic>? customFields;

  // Computed
  double get profitMargin => 
    sellingPrice > 0 ? ((sellingPrice - costPrice) / sellingPrice * 100) : 0;
  bool get isLowStock => currentStock <= minimumStock;
  bool get isExpiringSoon => expiryDate != null && 
    expiryDate!.difference(DateTime.now()).inDays <= 7;
  bool get isExpired => expiryDate != null && 
    expiryDate!.isBefore(DateTime.now());
  double get stockValue => currentStock * costPrice;
  double get stockPercentage => 
    maximumStock > 0 ? (currentStock / maximumStock * 100).clamp(0, 100) : 0;
}

class CircuitBreaker {
  String providerName;
  CircuitState state = CircuitState.closed;
  int failureCount = 0;
  int failureThreshold = 3;
  Duration resetTimeout = Duration(minutes: 5);
  DateTime? lastFailureTime;
  
  bool canAttempt():
    - if CLOSED → return true (healthy, allow request)
    - if OPEN → check if resetTimeout passed
      - if yes → set to HALF_OPEN, return true (test request)
      - if no → return false (skip this provider)
    - if HALF_OPEN → return true (allow test)
  
  void recordSuccess():
    - failureCount = 0
    - state = CLOSED
  
  void recordFailure():
    - failureCount++
    - lastFailureTime = now
    - if failureCount >= threshold → state = OPEN
}

class RateLimiter {
  Map<String, List<DateTime>> timestamps = {};
  
  bool canMakeRequest(String provider, int maxPerMinute):
    - Get timestamps for provider
    - Remove entries older than 60 seconds
    - Return timestamps.length < maxPerMinute
  
  void recordRequest(String provider):
    - Add DateTime.now() to timestamps[provider]
}

class AIFallbackManager {
  List<BaseAIProvider> providers;  // 8 providers in order
  Map<String, CircuitBreaker> circuitBreakers;
  RateLimiter rateLimiter;
  CostTracker costTracker;
  AILogger logger;
  
  Future<AIResponse> getResponse(AIRequest request):
    
    List<String> errors = [];
    Stopwatch stopwatch = Stopwatch()..start();
    
    FOR each provider in providers (ordered by priority):
      
      // CHECK 1: Circuit Breaker
      if circuitBreaker is OPEN → skip, add to errors
      
      // CHECK 2: Rate Limit
      if rate limited → try rotating key
        if no more keys → skip, add to errors
      
      // CHECK 3: Cost Limit (paid only)
      if provider.tier == PAID:
        if costTracker.totalToday >= dailyLimit → skip
        if settings.enablePaidFallback == false → skip
        LOG WARNING: "⚠️ Using PAID provider: {name}"
      
      // ATTEMPT REQUEST
      try:
        LOG: "🔄 Trying {provider.name} (Tier: {tier})..."
        
        response = await attemptWithRetry(provider, request)
        
        circuitBreaker.recordSuccess()
        rateLimiter.recordRequest()
        costTracker.recordCost()
        logger.logSuccess()
        
        LOG: "✅ {provider.name} succeeded ({latency}ms, ${cost})"
        
        return response with:
          - provider name
          - tier (local/free/paid)
          - cost
          - latency
          - fallbacksAttempted count
          - list of failed providers
      
      catch (error):
        circuitBreaker.recordFailure()
        errors.add("{provider.name}: {error}")
        LOG: "❌ {provider.name} failed: {error}"
        
        // Special: On rate limit, try key rotation
        if error is RateLimitError AND provider.hasMoreKeys:
          provider.rotateKey()
          try again once with new key
        
        continue to next provider
    
    // ALL 8 PROVIDERS FAILED
    return AIResponse(
      text: getOfflineFallbackResponse(request),
      provider: 'offline',
      tier: local,
      success: false,
      errorMessage: all errors joined
    )
